    <script>
        // ELIMINADO: Todo el motor 3D y variables de animación
        let currentStep = 0;
        let responses = [];

        const researchQuestions = [
            { q: "¿Cuál es el motor principal de conversión?", a: ["Seguridad económica / ROI", "Aspiración / Estatus Social", "Conveniencia / Ahorro de Tiempo", "Afinidad de Valores / Ética"] },
            { q: "¿Qué nivel de conocimiento tecnico tiene su usuario?", a: ["Básico: Requiere educación visual", "Medio: Compara estética y funciones", "Avanzado: Valora el minimalismo técnico", "Experto: Busca herramientas puras"] },
            { q: "¿Cuál es el mayor freno de compra identificado?", a: ["Falta de pruebas de autoridad", "Diseño percibido como genérico", "Exceso de pasos para convertir", "Desconfianza en el propósito"] },
            { q: "¿Cómo interactúa ante un cambio de precios?", a: ["Busca la opción más barata", "Acepta si el estatus se mantiene", "Analiza el beneficio técnico", "Sigue fiel por valores"] },
            { q: "¿En qué ecosistema digital habita?", a: ["Foros y contenido escrito denso", "Redes visuales (IG / TikTok)", "Plataformas B2B / LinkedIn", "Canales cerrados y exclusivos"] },
            { q: "¿Cuál es su miedo transaccional?", a: ["Haber perdido su inversión", "No verse exitoso frente a pares", "Que la solución sea ineficiente", "Que la marca sea una fachada"] },
            { q: "¿Qué tipo de estética consume?", a: ["Promocional y colorida", "Cinematográfica y pulida", "Cruda, técnica e industrial", "Minimalista con propósito"] },
            { q: "¿Qué cierra la venta finalmente?", a: ["Un descuento táctico", "Una referencia de autoridad", "Una especificación técnica clara", "Una historia de marca coherente"] }
        ];

        // NUEVO: Función para actualizar la barra de progreso
        function updateProgressBar() {
            const totalQuestions = researchQuestions.length;
            const percentage = (currentStep / totalQuestions) * 100;
            document.getElementById('progress-bar').style.width = percentage + '%';
        }

        function startInjection() {
            const b = document.getElementById('user-brand').value;
            const s = document.getElementById('user-sector').value;
            if(!b || !s) return alert("COMPLETE LA IDENTIFICACIÓN");
            
            document.getElementById('start-screen').classList.add('hidden');
            
            // NUEVO: Activar contenedor de progreso
            document.getElementById('progress-container').style.display = 'block';
            updateProgressBar(); // Iniciar en 0%

            // ELIMINADO: init3D(), animate() y timeouts de movimiento
            nextQuestion(); // Iniciar preguntas directamente
        }

        function nextQuestion() {
            if(currentStep >= researchQuestions.length) return endMission();
            
            const q = researchQuestions[currentStep];
            document.getElementById('modal').classList.remove('hidden');
            document.getElementById('question-text').innerText = q.q;
            
            const container = document.getElementById('options-container');
            container.innerHTML = '';
            q.a.forEach((opt, i) => {
                const btn = document.createElement('button');
                btn.className = 'option-btn'; btn.innerText = opt;
                btn.onclick = () => {
                    responses.push(i); currentStep++;
                    document.getElementById('modal').classList.add('hidden');
                    
                    // NUEVO: Actualizar barra tras responder
                    updateProgressBar();
                    
                    // Pequeño delay para sensación táctica antes de la siguiente pregunta
                    setTimeout(nextQuestion, 400); 
                };
                container.appendChild(btn);
            });
        }

        function endMission() {
            // NUEVO: Asegurar barra al 100%
            updateProgressBar();
            
            document.getElementById('result-screen').classList.remove('hidden');
            const brand = document.getElementById('user-brand').value.toUpperCase();
            const sector = document.getElementById('user-sector').value.toUpperCase();
            document.getElementById('out-brand').innerText = brand;
            document.getElementById('out-sector').innerText = sector;

            const names = ["EL ESTRATEGA DE RENDIMIENTO", "EL CURADOR DE ESTATUS", "EL PRAGMÁTICO INDUSTRIAL", "EL LÍDER DE COMUNIDAD"];
            const drivers = ["Eficiencia y ROI", "Validación y Prestigio", "Tiempo y Simplicidad", "Afinidad Ética"];

            document.getElementById('r-name').innerText = names[responses[0]];
            document.getElementById('r-driver').innerText = drivers[responses[0]];
            document.getElementById('r-digital').innerText = responses[4] == 1 ? "ALTA (Social-Driven)" : "MEDIA (Task-Driven)";
            document.getElementById('r-risk').innerText = responses[5] == 0 ? "ALTA (Aversión a la pérdida)" : "MODERADA (Busca ventaja)";
            document.getElementById('r-loyalty').innerText = responses[3] == 3 ? "TRIBAL (Fidelidad total)" : "TRANSACCIONAL (Utilidad)";
            document.getElementById('r-visual').innerText = responses[6] == 2 ? "INDUSTRIAL / CRUDA" : "PULIDA / PREMIUM";
            
            document.getElementById('r-diag').innerText = `INFORME GRAVA: EN EL SECTOR ${sector}, SU CLIENTE NO BUSCA UN PRODUCTO, BUSCA UNA CERTEZA TÉCNICA. LA IDENTIDAD DE ${brand} DEBE ABANDONAR LA PUBLICIDAD GENÉRICA Y POSICIONARSE COMO UN ARTEFACTO DE ALTA PRECISIÓN. MENOS ESFUERZO EN VENDER, MÁS ESFUERZO EN SER.`;
        }

        // ELIMINADO: EventListener de mousemove y función animate()
    </script>
